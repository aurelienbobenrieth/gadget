**What the linter reports:**

> Set returnType explicitly in options.

## Examples

<Tabs>
  <Tab title="Incorrect">

```ts title="api/models/widget/actions/create.ts"
export const options: ActionOptions = {
  // missing returnType - defaults differ between model and global actions
  timeoutMS: 50000,
};
```

  </Tab>
  <Tab title="Correct">

```ts title="api/models/widget/actions/create.ts"
export const options: ActionOptions = {
  timeoutMS: 50000,
  // set returnType explicitly to avoid confusion between model and global defaults
  returnType: false,
};
```

  </Tab>
